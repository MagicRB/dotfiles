#+title: whitespace.el
#+filetags: emacs-load

Highlight trailing whitespace.

#+begin_src emacs-lisp
    (use-package whitespace
      :config
      (setq whitespace-style '(face tabs trailing))
      (face-spec-set
       'whitespace-tabs
       '((t :background "red"))
       'face-defface-spec)
      (face-spec-set
       'whitespace-trailing
       '((t :background "red"))
       'face-defface-spec)
      (global-whitespace-mode))
#+end_src
