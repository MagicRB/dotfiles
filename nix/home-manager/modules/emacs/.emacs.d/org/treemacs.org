    :straight t
    :config
    (setq doom-themes-treemacs-theme "doom-colors") ; use the colorful treemacs theme

    :straight t)
  (use-package lsp-treemacs
    :straight t
    :after (lsp-mode treemacs)
    :config
    (lsp-treemacs-sync-mode 1))
