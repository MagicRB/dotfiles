     (use-package doom-modeline
       :straight t
       :config
       (doom-modeline-mode)
       (make-variable-buffer-local 'doom-modeline-icon)
       (add-hook 'after-make-frame-hook
		 (lambda ()
		   (setq doom-modeline-icon (display-graphic-p)))))
